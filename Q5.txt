#include <iostream>
using namespace std;
int main() {
    int i, n;
    bool p = true;
cout << "Enter a positive integer: ";
    cin >> n;
if (n == 0 || n == 1) {
        p = false;
    }
    else {
        for (i = 2; i <= n / 2; ++i) {
            if (n % i == 0) {
                p = false;
                break;
            }
        }
    }
    if (p)
        cout << n << " is a prime number";
    else
        cout << n << " is not a prime number";
}
